select s.state, s.city, s.pincode, sum(p.qty) from monet_storedim s, monet_purchasefact p where s.scode = p.scode group by s.state, s.city,s.pincode;
select s.sname, i.iname, d.purchasedate, p.qty from monet_storedim s, monet_itemdim i, monet_dateofpurchase d, monet_purchasefact p where s.scode = p.scode and i.icode = p.icode and d.dcode = p.dcode;
select s.city, i.category, sum(p.qty*i.price) from monet_storedim s,monet_itemdim i, monet_purchasefact p where s.scode = p.scode and i.icode = p.icode group by s.city, i.category;
select s.city, i.subcategory, d."month", sum(p.qty*i. price) from monet_storedim s, monet_itemdim i, monet_dateofpurchase d, monet_purchasefact p where s.scode = p.scode and i.icode=p.icode and d.dcode = p.dcode group by s.city,i.subcategory, d."month";
select s.state, s.city, s.pincode, i.category, i.subcategory, d."year", d.quarter, d."month",d.dayoftheweek,sum(p.qty*i. price) from monet_storedim s, monet_itemdim i, monet_dateofpurchase d, monet_purchasefact p where s.scode = p.scode and i.icode=p.icode and d.dcode = p.dcode group by s.state, s.city, s.pincode, i.category, i.subcategory, d."year", d.quarter, d."month", d.dayoftheweek;
